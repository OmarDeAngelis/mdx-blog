<h1 title="true">How to set a Simple React Counter</h1>

```jsx
const Counter = () => {
  const [value, setValue] = React.useState(0);
  React.useEffect(() => {
    const timer = setTimeout(() => {
      setValue(value + 1);
    }, 4000);
    return () => clearTimeout(timer);
  });

  return (
    <div>
      <h3> {value}</h3>
      <button
        style={{
          background: "red",
          borderRadius: "16px",
          padding: "0.2rem 0.8rem",
        }}
        onClick={() => setValue(0)}
      >
        {" "}
        Reset{" "}
      </button>
    </div>
  );
};
```
